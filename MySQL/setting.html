    # MYSQL 계정 생성 및 권한 부여

    실제 서비스에 사용되는 DB는 DBA가 root 권한을 가지고 있고, 계정을 역할에 맞도록 권한을 부여받아서 사용하도록 합니다

    # 표식 가이드

    
// 전체
// *

// 전체스키마.전체테이블
// *.*

// 내부접속만 가능
// localhost

// 외부접속가능
// %

    # DB 접속 및 계정 조회

    
// mysql root 권한 접속
mysql -u root -p

// 초기 비밀번호
0000

// mysql 스키마 선택
use mysql;

// 사용자 목록 조회
select user,host from user;

    # 사용자 생성

    
create user '사용자'@'host' identified by '비밀번호';

// ex1) 내부 접근을 허용하는 사용자 추가
create user 'test'@'localhost' identified by '0000';

// ex2) 외부 접근을 허용하는 사용자 추가
create user 'test'@'%' identified by '0000';

// ex3) 특정 ip만 접근을 허용하는 사용자 추가
create user 'test'@'123.456.789.100' identified by '0000';

// ex4) 특정 ip 대역을 허용하는 사용자 추가
create user 'test'@'192.168.%' identified by '0000';

    # 사용자 삭제

    

drop user '사용자';
// 또는
delete from user where user = '사용자';

// 예시
drop user 'test'@'localhost';

    # 사용자 권한 확인

    
// 사용자 권한을 확인
show grants for 'user'@'host';

    # 사용자 권한 부여

    
// 모든 데이터베이스의 모든 테이블에 모든 권한을 줌
grant all privileges on *.* to '사용자'@'localhost';

// 특정 데이터베이스의 모든 테이블에 모든 권한을 줌
grant all privileges on DB이름.* to '사용자'@'localhost';

// 특정 데이터베이스의 특정 테이블에 모든 권한을 줌
grant all privileges on DB이름.테이블명 to '사용자'@'localhost';

// 특정 데이터베이스의 특정 테이블에 select 권한을 줌
grant select on DB이름.테이블명 to '사용자'@'localhost';

// 특정 데이터베이스의 특정 테이블에 select, insert 권한을 줌
grant select, insert on DB이름.테이블명 to '사용자'@'localhost';

// 특정 데이터베이스의 특정 테이블에 컬럼1과 컬럼2의 update 권한을 줌
grant update(컬럼1, 컬럼2) on DB이름.테이블명 to '사용자'@'localhost';

    # 여러 권한 부여 및 제한

    
// 권한 부여
grant select, insert, update on mydb.* to 'test'@'localhost';

// 권한 제거
revoke select, insert, update on mydb.* from 'test'@'localhost';

    # 특정 테이블에 대한 권한 제한

    
// 특정 테이블에 권한 부여
grant select on mydb.mytable to 'test'@'localhost';

// 특정 테이블에 권한 제거
revoke select on mydb.mytable from 'test'@'localhost';

    # 모든 권한 제거

    
revoke all privileges on *.* from 'user'@'localhost';

    # 권한 즉시 반영

    
FLUSH PRIVILEGES;

    권한을 반영해주어야 실행된 명렁에대한 변화를 확인할 수 있을 것입니다.

    # 외부접속 허용시 주의사항

    DB 기본 포트는 3306

    초기 유저 이름은 root

    초기 기본 비밀번호는 0000

    해당 기본값 정보들을 유지한상태로 사용하면 봇들의 접속시도에 위험을 받게 됩니다.

    모든 스키마와 테이블에 대한 권한을 가진 유저가 제일 쉬운 접근에 노출되면

    훗날 모든 스키마를 drop 당하는 사태가 발생할 수 있으니

    포트, 유저, 비밀번호는 꼭 다른 정보로 수정하여 사용하는것을 권장합니다.
